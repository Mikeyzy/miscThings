	IN
	STO r0
	IN
	STO base
	STO r2
	STO r3

sStart	LDA zero
	STO r1
sLoop	ADD base
	STO square
	LDA r3
	SUB one
	STO r3
	BRZ sReturn
	LDA square
	BR  sLoop
sReturn	LDA r2
	BRZ x3	#go to 'x3' if zero

digit3	LDA r0
	SUB hund
	BRP saveD3
digit2	LDA r0
	SUB ten
	BRP saveD2
digit1	LDA r0
	ADD r1
	STO r0
	BR  toBase
saveD3	STO r0
	LDA r1
	ADD square
	BR  dLoop
saveD2	STO r0
	LDA r1
	ADD base
dLoop	STO r1
	BR  digit3

toBase	IN
	STO base
	STO r3
	LDA zero
	STO r2
	BR  sStart	#call s Function(square)

x3	LDA r0
	SUB square
	BRP saveX3
x2	LDA r0
	SUB base
	BRP saveX2
x1	LDA r0
	ADD r1
	BR  output
saveX3	STO r0
	LDA r1
	ADD hund
	BR  xLoop
saveX2	STO r0
	LDA r1
	ADD ten
xLoop	STO r1
	BR  x3

#variables
r0	DAT 0
r1	DAT 0
r2	DAT 0
r3	DAT 0
base	DAT 0
square	DAT 0
#constants
one	DAT 1
two	DAT 2
ten	DAT 10
hund	DAT 100
inf	DAT 999
output	OUT
zero	HLT
